PROGRAM:
    FUNCTION * // While not EOF Keep Scanning

FUNCTION:
    TYPE IDENTIFIER L_PAR PARAM_LIST R_PAR STATEMENT

TYPE:
    INT|FLOAT|BOOL|STRING|CHAR

PARAM_LIST:
    TYPE IDENTIFIER 
    | TYPE IDENTIFIER [COMMA TYPE IDENTIFIER]* // While next char not R_PAR keep scanning

ARG_LIST:
    EXPERSSION [COMMA EXPERSSION]* // While next token not R_PAR keep scanning

STATEMENT:
    L_BRC STATEMENT* R_BRC // While next token not R_BRC keep scanning 
    RETURN EXPRESSION SEMI
    IDENTIFIER ASSIGN EXPRESSION SEMI
    TYPE IDENTIFIER ASSIGN EXPRESSION SEMI
    IF L_PAR EXPERSSION R_PAR STATEMENT
    IF L_PAR EXPERSSION R_PAR STATEMENT ELSE STATEMENT
    WHILE L_PAR EXPERSSION R_PAR STATEMENT
    SEMI

EXPRESSION2:
    LITERAL
    L_PAR EXPERSSION R_PAR
    IDENTIFIER L_BKT EXPERSSION R_BKT
    UNARY_OP EXPRESSION
    IDENTIFIER ARG_LIST

EXPRESSION:
    EXPRESSION2 
    EXPERSSION2 BINARY_OP EXPRESSION2

LITERAL:
    INT
    FLOAT
    DOUBLE
    STRING
    CHAR

UNARY_OP:
    ADD
    SUB
    NOT
    BNOT

BINARY_OP:
    ADD
    SUB
    MUL
    DIV
    MOD
    
    OR
    AND
    EQ
    LT
    GEQ
    LEQ
    GT
    NEQ
    BOR
    BAND
    BNOT
    BXOR
